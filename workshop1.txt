pipeline {
    agent any

    stages {
        stage('Pull Docker Image') {
            steps {
                script {
                    sh '''docker pull chontipanpepe/my-image:main'''
                }
            }
        }
        stage('Run Docker Container') {
            steps {
                script {
                    sh '''docker run --rm -p 3000:3000 --name se67 -d  chontipanpepe/my-image:main'''
                }
            }
        }
    }
}
